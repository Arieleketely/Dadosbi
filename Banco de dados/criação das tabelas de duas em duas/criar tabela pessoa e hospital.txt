-- Criação da tabela Medicamento
IF NOT EXISTS (
    SELECT 1 FROM sys.tables WHERE name = 'Hospital'
)
BEGIN
    CREATE TABLE Hospital (
        id_hospital INT IDENTITY(1,1) PRIMARY KEY,
       nome VARCHAR(255) NOT NULL,
        estado VARCHAR(255),
        cidade VARCHAR(255),
        bairro VARCHAR(255),
        numero INT,
        complemento VARCHAR(255),
         telefone VARCHAR(255)
    );
END;
GO

-- Criação da tabela Plano
IF NOT EXISTS (
    SELECT 1 FROM sys.tables WHERE name = 'Pessoa'
)
BEGIN
    CREATE TABLE Pessoa (
        id_pessoa INT IDENTITY(1,1) PRIMARY KEY,
        nome VARCHAR(255) NOT NULL,
        estado VARCHAR(255),
        cidade VARCHAR(255),
        bairro VARCHAR(255),
        numero INT,
        complemento VARCHAR(255),
        telefone VARCHAR(255)

    );
END;
GO
